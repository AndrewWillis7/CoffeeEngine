#include "Master.h"
#include <gdiplus.h>
#pragma comment(lib, "gdiplus.lib")

using namespace Gdiplus;
ULONG_PTR gdiToken;

Master::Master(HINSTANCE hInstance) :
    m_hInstance(hInstance), m_isRunning(false) {}

Master::~Master() {
    GdiplusShutdown(gdiToken);
}

void Master::CalculateInitialTaskbarPosition(int& x, int& y) {
    RECT workingArea;
    SystemParametersInfo(SPI_GETWORKAREA, 0, &workingArea, 0);

    x = (workingArea.right - workingArea.left) - (baseWidth * m_pet->GetScale()) - 40;
    y = (workingArea.bottom - workingArea.top) - (baseHeight * m_pet->GetScale()/2);

    //x = 300;
    //y = 300;
}

bool Master::Initialize() {
    GdiplusStartupInput gdiStartupInput;
    GdiplusStartup(&gdiToken, &gdiStartupInput, NULL);

    m_pet = std::make_unique<Pet>(baseWidth, baseHeight);
    m_window = std::make_unique<Window>(m_hInstance, (int)(baseWidth * m_pet->GetScale()), (int)(baseHeight * m_pet->GetScale()));

    RECT workingArea;
    SystemParametersInfo(SPI_GETWORKAREA, 0, &workingArea, 0);

    m_pet->SetBounds(workingArea.left + 40, workingArea.right - 40);

    int startX, startY;
    CalculateInitialTaskbarPosition(startX, startY);
    m_pet->SetPosition(startX, startY);

    return m_window->Create(L"Desktop Pet Engine", startX, startY);
}

void Master::Run() {
    m_isRunning = true;
    MSG msg = {};
    DWORD lastTime = GetTickCount();

    

    // Setting the FPS timer
    SetTimer(m_window->GetHandle(), 2, 16, NULL);

    while (m_isRunning) {
        if (PeekMessage(&msg, NULL, 0, 0, PM_REMOVE)) {
            if (msg.message == WM_QUIT) {
                m_isRunning = false;
                break;
            }
            TranslateMessage(&msg);
            DispatchMessage(&msg);
        }

        if (!m_isRunning) break;

        // .. Core Game loop here: Execution step
        DWORD currentTime = GetTickCount();
        float deltaTime = (currentTime - lastTime) / 1000.0f;
        lastTime = currentTime;

        m_pet->update(deltaTime);
        m_window->UpdatePosition(m_pet->GetX(), m_pet->GetY());

        m_window->Render(*m_pet);

        // Limit the loop cycle frequency to optimize CPU Overhead
        Sleep(16);
        
    }
}
